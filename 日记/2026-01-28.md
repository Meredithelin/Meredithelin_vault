 <%*
// 1. 获取基准日期（增加防报错逻辑）
// 如果文件名是"日记模板"这种文字，就强制用"今天"，防止报错
let title = tp.file.title;
if (!moment(title, "YYYY-MM-DD", true).isValid()) {
    title = tp.date.now("YYYY-MM-DD");
}

// 2. 计算相关日期
const tDate = moment(title, "YYYY-MM-DD");
const yesterday = tDate.clone().subtract(1, 'day').format("YYYY-MM-DD");
const tomorrow = tDate.clone().add(1, 'day').format("YYYY-MM-DD");

// 3. 输出日期导航链接
tR += `- **昨天**:: [[${yesterday}]]\n`
tR += `- **明天**:: [[${tomorrow}]]\n`
tR += `\n---\n`

// ========================================================
// 4. 获取昨天的未完成任务
// 【重要】请确认你的日记文件夹名字是 "日记" 还是 "Daily"？
// 根据你之前的截图，你的文件夹叫 "日记"，如果改了请在这里修改：
const folderName = "日记"; 
// ========================================================

// 拼凑昨天日记的完整路径
const path = folderName ? `${folderName}/${yesterday}.md` : `${yesterday}.md`;
const file = app.vault.getAbstractFileByPath(path);

// 读取文件并提取任务
if (file) {
    const content = await app.vault.read(file);
    // 正则：查找 "- [ ]" 开头的行（未完成任务）
    const incompleteTasksRegex = /^\s*- \[ \].+$/gm;
    const tasks = content.match(incompleteTasksRegex);

    if (tasks && tasks.length > 0) {
        tR += "### ⚡️ 昨日未竟之事\n" + tasks.join('\n');
    } else {
        tR += "✅ 昨日事昨日毕！";
    }
} else {
    // 如果找不到昨天的日记（比如这是你写的第一篇，或者昨天没写）
    tR += `🌴 未找到昨日日志 ([[${yesterday}]])，轻装上阵！`;
}
