<%*
// 1. 获取文件名作为日期
let title = tp.file.title;

// 2. 容错：如果文件名不是日期（比如你在测试时），就强制用今天，防止报错
if (!moment(title, "YYYY-MM-DD", true).isValid()) {
    title = tp.date.now("YYYY-MM-DD");
}

// 3. 计算日期变量
const tDate = moment(title, "YYYY-MM-DD");
const yesterday = tDate.clone().subtract(1, 'day').format("YYYY-MM-DD");
const lastYear = tDate.clone().subtract(1, 'year').format("YYYY-MM-DD");
const tomorrow = tDate.clone().add(1, 'day').format("YYYY-MM-DD");
const nextYear = tDate.clone().add(1, 'year').format("YYYY-MM-DD");

// 4. 输出内容（用 tR 输出是为了解决变量作用域问题）
tR += `- **昨天**:: [[${yesterday}]]\n`
tR += `- **去年今日**:: [[${lastYear}]]\n`
tR += `- **明天**:: [[${tomorrow}]]\n`
tR += `- **明年今日**:: [[${nextYear}]]`
%>
