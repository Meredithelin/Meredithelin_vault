<%*
// 1. 获取标题
let title = tp.file.title;

// 2. 检查标题是不是日期格式（为了防止在"日记模板"这个文件中测试时出错）
// 如果标题不是有效日期，默认使用"今天"作为基准
if (!moment(title, "YYYY-MM-DD", true).isValid()) {
    title = tp.date.now("YYYY-MM-DD");
}

// 3. 计算日期
const today = moment(title, "YYYY-MM-DD");
const yesterday = today.clone().subtract(1, 'day').format("YYYY-MM-DD");
const tomorrow = today.clone().add(1, 'day').format("YYYY-MM-DD");
const lastYear = today.clone().subtract(1, 'year').format("YYYY-MM-DD");
const nextYear = today.clone().add(1, 'year').format("YYYY-MM-DD");

// 4. 输出内容 (使用 tR += 来追加文本)
tR += `• 昨天:: [[${yesterday}]]\n`
tR += `• 去年今日:: [[${lastYear}]]\n`
tR += `• 明天:: [[${tomorrow}]]\n`
tR += `• 明年今日:: [[${nextYear}]]`
%>
