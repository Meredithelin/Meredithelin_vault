```c
#include <stdio.h>
#include <string.h>

#define MAX_TASKS 100

typedef struct {
    int id;
    char title[100];
    int priority;
    int is_done;
} Task;

Task todo_list[MAX_TASKS];
int task_count = 0;

void add_task(char *title, int priority) {
    if (task_count < MAX_TASKS) {
        todo_list[task_count].id = task_count + 1;
        strcpy(todo_list[task_count].title, title);
        todo_list[task_count].priority = priority;
        todo_list[task_count].is_done = 0;
        task_count++;
        printf("✅ 任务添加成功！\n");
    }
}

void show_tasks() {
    printf("\n--- 任务清单 ---\n");
    for (int i = 0; i < task_count; i++) {
        printf("[%d] %s | 优先级: %d | 状态: %s\n", 
               todo_list[i].id, todo_list[i].title, 
               todo_list[i].priority, 
               todo_list[i].is_done ? "完成" : "未完成");
    }
}

int main() {
    add_task("学习 C 语言指针", 3);
    add_task("整理 Obsidian 插件", 2);
    show_tasks();
    return 0;
}
```
